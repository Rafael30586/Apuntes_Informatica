docker images --> muestra la lista de las imágenes

dokcer pull <imagen>:<versión> --> descarga la imagen de la versión que indiquemos. Sin no aclaramos la versión descargará la última

docker image rm <imagen>:<versión> --> elimina la imagen que indiquemos

docker create <imagen>:<versión> --> crea un contenedor a partir de la imagen que indiquemos

docker create --name <nombre> <imagen>:<versión> --> crea un contenedor con el nombre que indiquemos

docker start <id_o_nombre_de contenedor> --> ejecuta el contenedor según el id que indiquemos

docker ps --> muestra todos los contenedores en ejecución

docker stop <id_de_contenedor> --> detiene el contenedor según el id que indiquemos

docker ps -a --> muestra todos los contenedores, incluidos los que no están en ejecución

docker rm <id_o_nombre_de contenedor> --> borra el contenedor según el nombre o id que indiquemos 

docker create -p<puerto_anfitrión>:<puerto_contenedor> <imagen> --> crea una contenedor de la imagen que indiquemos haciendo un mapeo con el puerto de la máquina anfitrión y el del contenedor según indiquemos. Si colocamos solo un puerto en lugar de dos, docker asignará un puerto del anfitrión al azar

docker logs <contenedor> --> muestra el registro del contenedor que indiquemos

docker logs --folow <contenedor> --> muestra los registros de un contenedor y permanece a la escucha de los mismos

docker run <imagen> --> descarga la imagen que indiquemos (si es que no la tenemos descargada ya), crea un contenedor a partir de la imagen y lo ejecuta. Se le puede agregar la opción --name <nombre> para darle nombre al contenedor. Agregando la opción -d hace que el contenedor se ejecute en segundo plano 

docker <comando> --help --> muestra ayuda del comando que le indiquemos 

docker exec --> sirve para ejecutar comandos en un contenedor en ejecución

docker network ls --> lista todas las redes

docker network create <nombre> --> crea una red con el nombre que le indiquemos

docker network rm <red> --> borra una red que indiquemos

docker create --network <red> <imagen> --> crea un contenedor asignándole la red que indiquemos

docker create -e <variable>=<valor> <imagen> --> crea un contenedor asignándole un valor a la variable de entorno que indiquemos

docker build -t <nombre>:<versión> <ruta> --> crea una imagen a partir del dockerfile presente en le ruta que indiquemos con el nombre que indiquemos y su versión. Si estamos posicionados en el mismo directorio donde está el dockerfile, el argumento sería un punto
